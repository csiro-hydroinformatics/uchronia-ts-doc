        template<typename T>
        bool AreEqual(const vector<T>& a, const vector<T>& b, bool strict = false, double tolerance = 1e-12)
        {
            size_t lengthA = a.size();
            size_t lengthB = b.size();

            if (lengthA != lengthB)
                return false;

            for (size_t i = 0; i < lengthA; i++)
            {
                auto valA = a[i];
                auto valB = b[i];

                if (strict && (valA != valB))
                    return false;
                else if (std::abs(valA - valB) > tolerance)
                    return false;
            }

            return true;
        }
